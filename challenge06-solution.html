duration_hms <- function(x) {
  s <- round(as.numeric(x))
  sprintf(
    "%02d:%02d:%02d",
    s %/% 3600,
    (s %% 3600) %/% 60,
    s %% 60
  )
}

visitduration_desc[c("Mean", "MAD", "Median", "SD")] <-
